#include <stdio.h>

// 정수를 32비트 2진수로 출력하는 함수
void printBinary(int n) {
    for (int i = 31; i >= 0; i--) {
        int bit = (n >> i) & 1; // i번째 비트를 추출
        printf("%d", bit);
        if (i % 8 == 0 && i != 0) printf(" "); // 8비트마다 공백 추가 (가독성)
    }
}

int main() {
    int a, b;

    printf("두 정수를 입력하세요 (예: 10 5): ");
    scanf("%d %d", &a, &b);

    printf("\n[입력값 확인]\n");
    printf("A: %10d | Binary: ", a); printBinary(a); printf("\n");
    printf("B: %10d | Binary: ", b); printBinary(b); printf("\n");

    printf("\n[비트 연산 결과]\n");

    // AND 연산
    printf("A & B : %7d | ", a & b); printBinary(a & b); printf("\n");
    // OR 연산
    printf("A | B : %7d | ", a | b); printBinary(a | b); printf("\n");
    // XOR 연산
    printf("A ^ B : %7d | ", a ^ b); printBinary(a ^ b); printf("\n");
    // NOT 연산 (A 기준)
    printf("~A    : %7d | ", ~a); printBinary(~a); printf("\n");
    // Left Shift (A를 2비트 이동)
    printf("A << 2: %7d | ", a << 2); printBinary(a << 2); printf("\n");
    // Right Shift (A를 2비트 이동)
    printf("A >> 2: %7d | ", a >> 2); printBinary(a >> 2); printf("\n");

    return 0;
}
